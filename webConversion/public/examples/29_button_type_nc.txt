// Button Type: NC (Normally Closed)
// ===================================
// A NC (Normally Closed) button has INVERTED logic.
// Default state: TRUE (not pressed)
// When pressed: FALSE
// When released: Returns to TRUE
//
// Instructions:
// 1. In the Default Scene, right-click I0.0 until it shows "NC"
// 2. I0.0 will show the PI button icon (circle with line)
// 3. Notice I0.0 starts at TRUE (1) without pressing
// 4. Press I0.0 - it becomes FALSE (0)
// 5. Release I0.0 - it returns to TRUE (1)
//
// This is useful for:
// - STOP buttons (fail-safe: wire break = stop)
// - Emergency stop (E-STOP) circuits
// - Safety interlock switches

// NC button controls output (inverted)
// Output is ON until you press the button
LD I0.0       // Load NC button (default TRUE)
OUT Q0.0      // Output ON when button NOT pressed

// START/STOP circuit with NC stop button
// I0.1 (NO) = START, I0.0 (NC) = STOP
LD I0.1       // Load START button (NO)
OR Q0.1       // OR with output (seal-in)
AND I0.0      // AND with STOP button (NC - breaks seal when pressed)
OUT Q0.1      // Motor control output

// Safety circuit: NC buttons in series
// Both NC buttons must be closed (not pressed) for output
// Pressing either button stops the output
LD I0.2       // Load NC safety button 1
AND I0.3      // AND with NC safety button 2
OUT Q0.2      // Output ON only when both buttons NOT pressed

// Emergency stop with auto-restart prevention
// I0.4 (NC) = E-STOP, I0.5 (NO) = RESET
LD M0         // Load run memory bit
AND I0.4      // AND with E-STOP (NC - normal operation)
OUT M0        // Maintain run state
OUT Q0.3      // Control output

// Reset/start circuit
LD I0.5       // Load RESET/START button (NO)
ANDN Q0.3     // Only if not already running
OUT M0        // Set run memory bit

// Button Type: NO (Normally Open)
// =================================
// A NO (Normally Open) button is a momentary push button.
// It is TRUE only while you hold it down.
// When you release it, it immediately returns to FALSE.
//
// Instructions:
// 1. In the Default Scene, right-click I0.0 until it shows "NO"
// 2. I0.0 will show the push button icon (not the switch)
// 3. Hold down I0.0 - it activates (TRUE)
// 4. Release I0.0 - it immediately deactivates (FALSE)
// 5. Q0.0 is ON only while you hold the button
//
// This is useful for:
// - START buttons (momentary activation)
// - Test/jog buttons
// - Manual override controls

// Direct push button control
LD I0.0       // Load NO push button (I0.0 as NO type)
OUT Q0.0      // Output ON only while button pressed

// Push button with seal-in (latching)
// Press I0.0 (NO) to start, I0.1 (NO) to stop
LD I0.0       // Load START button (NO)
OR Q0.1       // OR with output (seal-in)
ANDN I0.1     // AND NOT with STOP button (NO)
OUT Q0.1      // Latched output

// Push button counter
// Each press increments counter
LD I0.2       // Load push button (NO)
CTU C0, 5     // Count up to 5 presses
LD C0         // Load counter done bit
OUT Q0.2      // Light when 5 presses reached

// Reset counter with another button
LD I0.3       // Load reset button (NO)
CTR C0        // Reset counter when pressed
